    setterPrefix = "with",
    builderClassName = "Builder",
    buildMethodName = "build0",
    toBuilder = true
)
@Accessors(
    fluent = true
)

  private String codePart1 = "RG10";

  @lombok.Builder.Default
  private String codePart2 = "1LG";

  public static class Builder {

      return this.build0();

  }
}
